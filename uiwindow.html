UIWindow

创建一个UIButton控件示例

- (void)viewDidLoad
{
    [super viewDidLoad];

    // 创建UIWindow
    UIWindow *window1 = [[UIWindow alloc] init];

    // 设置UIWindow的位置
    window1.frame = CGRectMake(0, 0, 320, 300);

    // 设置UIWindow的背景颜色
    window1.backgroundColor = [UIColor redColor];

    // 显示UIWindow
    window1.hidden = NO;


    // 创建UIWindow
    UIWindow *window2 = [[UIWindow alloc] init];

    // 设置UIWindow的位置
    window2.frame = CGRectMake(0, 0, 320, 200);

    // 设置UIWindow的背景颜色
    window2.backgroundColor = [UIColor blueColor];

    // 设置UIWindow的级别
    window2.windowLevel = UIWindowLevelStatusBar;

    // 显示UIWindow
    window2.hidden = NO;


    // 创建UIWindow
    UIWindow *window3 = [[UIWindow alloc] init];

    // 设置UIWindow的位置
    window3.frame = CGRectMake(0, 0, 320, 100);

    // 设置UIWindow的背景颜色
    window3.backgroundColor = [UIColor grayColor];

    // 设置UIWindow的级别
    window3.windowLevel = UIWindowLevelAlert;

    // 显示UIWindow
    window3.hidden = NO;


    // 打印三个UIWindow的级别
    NSLog(@"window1 level %f", window1.windowLevel);
    NSLog(@"window2 level %f", window2.windowLevel);
    NSLog(@"window3 level %f", window3.windowLevel);

    [self.view addSubview:window1];
    [self.view addSubview:window2];
    [self.view addSubview:window3];
}
