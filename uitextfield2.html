####UITextField事件绑定 UITextFieldDelegate ######是否允许进入编辑模式 - (BOOL)textFieldShouldBeginEditing:(UITextField *)textField { return YES; } ######进入编辑模式 - (void)textFieldDidBeginEditing:(UITextField *)textField { NSLog(@"进入编辑模式"); } ######是否可以退出编辑模式 NO:不可切换textfield - (BOOL)textFieldShouldEndEditing:(UITextField *)textField { return YES; } ######退出编辑模式 - (void)textFieldDidEndEditing:(UITextField *)textField { NSLog(@"退出了编辑模式"); } ######是否可以点击清除按钮 - (BOOL)textFieldShouldClear:(UITextField *)textField { return NO; } ######是否可以点击清除按钮 - (BOOL)textFieldShouldReturn:(UITextField *)textField { [textField resignFirstResponder]; // 收起键盘方法 return YES; } ######键盘升起通知 [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(keyboardWillShow) name:UIKeyboardWillShowNotification object:nil]; // 执行方法 - (void)keyboardWillShow { NSLog(@"键盘升起来了"); }

UITextField事件绑定

UITextFieldDelegate
是否允许进入编辑模式
- (BOOL)textFieldShouldBeginEditing:(UITextField *)textField
{
    return YES;
}
进入编辑模式
- (void)textFieldDidBeginEditing:(UITextField *)textField
{
    NSLog(@"进入编辑模式");
}
是否可以退出编辑模式 NO:不可切换textfield
- (BOOL)textFieldShouldEndEditing:(UITextField *)textField
{
    return YES;
}
退出编辑模式
- (void)textFieldDidEndEditing:(UITextField *)textField
{
    NSLog(@"退出了编辑模式");
}
是否可以点击清除按钮
- (BOOL)textFieldShouldClear:(UITextField *)textField
{
    return NO;
}
是否可以点击清除按钮
- (BOOL)textFieldShouldReturn:(UITextField *)textField
{
    [textField resignFirstResponder]; // 收起键盘方法
    return YES;
}
键盘升起通知
[[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(keyboardWillShow) name:UIKeyboardWillShowNotification object:nil];

// 执行方法
- (void)keyboardWillShow
{
    NSLog(@"键盘升起来了");
}
