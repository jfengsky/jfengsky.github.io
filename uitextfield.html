UITextField

创建一个UITextField控件示例

- (void)viewDidLoad
{
    [super viewDidLoad];

    // 默认textfield是透明的
    UITextField *textField = [[UITextField alloc] init];
    textField.frame = CGRectMake(10, 30, 300, 30);

    // 设置边框样式
    textField.borderStyle = UITextBorderStyleRoundedRect;

    // 设置背景颜色
    textField.backgroundColor = [UIColor blueColor];

    // 设置提示文字
    textField.placeholder = @"请输入密码";

    // 设置密文输入
    textField.secureTextEntry = YES;

    // 设置键盘样式 数字键盘等
    textField.keyboardType = UIKeyboardTypeNumberPad;

    // 设置键盘风格
    textField.keyboardAppearance = UIKeyboardAppearanceDark;

    // 设置弹出视图
    UIView *tempView = [[UIView alloc] init];
    tempView.backgroundColor = [UIColor blackColor];
    tempView.frame = CGRectMake(0, 100, 320, 100);
    textField.inputView = tempView;

    // 设置左视图
    UIView *leftView = [[UIView alloc] init];
    leftView.frame = CGRectMake(0, 0, 100, 100);
    leftView.backgroundColor = [UIColor orangeColor];
    textField.leftView = leftView;

    // 设置左视图模式 坐标不起作用，只有宽和高起作用
    textField.leftViewMode = UITextFieldViewModeAlways;

    // 设置清除按钮模式
    textField.clearButtonMode = UITextFieldViewModeAlways;

    // 再次编辑是否清空之前内容
    // textField.clearsOnBeginEditing = YES;

    // 内容纵向对齐方式
    textField.contentVerticalAlignment = UIControlContentVerticalAlignmentCenter;

    // 内容横向对齐方式
    textField.contentHorizontalAlignment = UIControlContentVerticalAlignmentCenter;

    // 文字内容对齐方式
    textField.textAlignment = NSTextAlignmentCenter;

    // 设置textfield滚动 文字会缩小
    textField.adjustsFontSizeToFitWidth = YES;

    // 设置最小字号，与textfield滚动相关
    textField.minimumFontSize = 50;

    // 设置字母是否大写
    textField.autocapitalizationType = UITextAutocapitalizationTypeAllCharacters;

    // 设置return样式
    textField.returnKeyType = UIReturnKeyGoogle;

    [self.view addSubview:textField];
}
